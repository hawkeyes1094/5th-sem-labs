# For the DNS clients
# DNS clients are R2, R3, R4
conf t
inter f0/0
ip address 10.0.10.1X 255.255.255.0
no shut
do wr
ip domain lookup
ip name-server 10.0.10.1
ip route 0.0.0.0 0.0.0.0 10.0.10.1
end
ping loopback.R1.com
#

#For the DNS server
inter f0/0
ip address 10.0.10.1 255.255.255.0
no shut
do wr
exit

ip dns server
ip host loopback.R1.com 2.2.2.2
inter loopback 1
ip address 2.2.2.2 255.255.255.255
end

ping loopback.R1.com
#
